### python包相关
#### 在线安装
已安装python包查看  </br>
''' bash
 '''
pip list
env/pip list
安装库   </br>
pip install packagename
更新包   </br>
pip install --upgrade packagename
不适用缓存更新包    </br>
pip install --upgrade packagename --no-cache
更新pip   </br>
python -m pip install -- upgrade pip
批处理安装   </br>
pip install -r requirements.txt
requirements.txt like:
    jupyter==1.0.0
    Keras==2.1.3
    numpy==1.15.1
    pandas==0.22.0
    scikit-learn==0.19.1
